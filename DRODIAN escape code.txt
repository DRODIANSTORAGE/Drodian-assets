      function getCurrentKillCapacity() {
        let total = 0;
        // Knife: 2 kills when collected
        if (knifeRemaining > 0) total += knifeRemaining;
        // Sneek free kill
        total += gameState.freeKillsRemaining || 0;
        // Gun: +3 when collected
        if (gunPerk && gunPerk.collected) total += gunPerk.shotsRemaining || 0;
        // Second gun: +3 when collected
        if (gunPerk2 && gunPerk2.collected) total += gunPerk2.shotsRemaining || 0;
        // Active ally kills
        if (ally && ally.active) total += ally.killsRemaining || 0;
        // Queued allies
        if (allyQueue && allyQueue.length) {
          allyQueue.forEach((helper) => {
            total += helper.killsRemaining || 0;
          });
        }
        // Black D: +3 potential kills when not collected
        if (blackD && !blackD.collected) total += 3;
        return total;
      }

      function playTone(freq, duration, type = 'square', volume = 0.03) {
        if (muted) return;
        try {
          audioCtx = audioCtx || new (window.AudioContext || window.webkitAudioContext)();
          const osc = audioCtx.createOscillator();
          const gain = audioCtx.createGain();
          osc.type = type;
          osc.frequency.value = freq;
          gain.gain.value = volume;
          osc.connect(gain);
          gain.connect(audioCtx.destination);
          osc.start();
          gain.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + duration);
          osc.stop(audioCtx.currentTime + duration);
        } catch (e) {}
      }

      function playSoundEffect(url, volume = 0.5) {
        if (muted || !url) return;
        try {
          const sfx = new Audio(url);
          sfx.volume = volume;
          sfx.preload = 'auto';
          sfx.crossOrigin = 'anonymous';
          sfx.play().catch(() => {});
        } catch (e) {}
      }

      // Particle system functions
      function createParticle(x, y) {
        let particle;
        if (particlePool.length > 0) {
          particle = particlePool.pop();
        } else if (particles.length < maxParticles) {
          const el = document.createElement('div');
          el.className = 'particle';
          el.style.cssText = 'position: absolute; width: 6px; height: 6px; background: #8a8a8a; border-radius: 1px; pointer-events: none; z-index: 10;';
          world.appendChild(el);
          particle = { el };
        } else {
          return null;
        }
        
        particle.x = x;
        particle.y = y;
        particle.vx = (Math.random() - 0.5) * 80;
        particle.vy = (Math.random() - 0.5) * 80;
        particle.life = 0.5; // 500ms
        particle.maxLife = 0.5;
        particle.size = 4 + Math.random() * 4;
        
        particles.push(particle);
        return particle;
      }

      function updateParticles(dt) {
        for (let i = particles.length - 1; i >= 0; i--) {
          const p = particles[i];
          p.life -= dt;
          
          if (p.life <= 0) {
            p.el.style.display = 'none';
            particlePool.push(p);
            particles.splice(i, 1);
            continue;
          }
          
          // Update position
          p.x += p.vx * dt;
          p.y += p.vy * dt;
          
          // Fade out
          const alpha = p.life / p.maxLife;
          p.el.style.opacity = alpha * 0.6;
          p.el.style.display = 'block';
        }
      }

      function renderParticles() {
        particles.forEach(p => {
          p.el.style.left = (offsetX + p.x * scale) + 'px';
          p.el.style.top = (offsetY + p.y * scale) + 'px';
          p.el.style.transform = 'translate(-50%, -50%) scale(' + scale + ')';
        });
      }

      function shuffle(array) {
        for (let i = array.length - 1; i > 0; i--) {
          const j = Math.floor(Math.random() * (i + 1));
          [array[i], array[j]] = [array[j], array[i]];
        }
        return array;
      }

      function generateLetterPositions(count) {
        const positions = [];
        const minDistance = 130;
        const maxAttempts = 400;
        const safeMargin = 90;

        for (let i = 0; i < count; i++) {
          let placed = false;
          for (let attempt = 0; attempt < maxAttempts; attempt++) {
            const candidate = {
              x: safeMargin + Math.random() * (WORLD_W - safeMargin * 2),
              y: 220 + Math.random() * (WORLD_H - 420)
            };
            const tooCloseToPlayer = Math.hypot(candidate.x - 500, candidate.y - 180) < 180;
            const collidesWall = (levelWalls[currentLevel] || []).some((wall) => {
              const halfW = wall.w * 0.5 + 36;
              const halfH = wall.h * 0.5 + 36;
              return candidate.x > wall.x - halfW && candidate.x < wall.x + halfW && candidate.y > wall.y - halfH && candidate.y < wall.y + halfH;
            });
            const collidesSwamp = (levelSwamps[currentLevel] || []).some((swamp) => {
              const halfW = swamp.w * 0.5 + 28;
              const halfH = swamp.h * 0.5 + 28;
              return candidate.x > swamp.x - halfW && candidate.x < swamp.x + halfW && candidate.y > swamp.y - halfH && candidate.y < swamp.y + halfH;
            });
            const tooClose = positions.some((pos) => Math.hypot(candidate.x - pos.x, candidate.y - pos.y) < minDistance);
            if (!tooCloseToPlayer && !collidesWall && !collidesSwamp && !tooClose) {
              positions.push(candidate);
              placed = true;
              break;
            }
          }
          if (!placed) {
            positions.push({
              x: safeMargin + ((i * 97) % (WORLD_W - safeMargin * 2)),
              y: 220 + ((i * 143) % (WORLD_H - 420))
            });
          }
        }

        return shuffle(positions);
      }

      function playLevelMusic(levelIndex) {
        // Stop intense music when playing level music
        stopVictoryMusic();
        
        // Check if we're already playing the correct music for this level
        if (musicElement && currentMusic === levelIndex && !musicElement.paused) {
          updateMusicDebug(levelIndex, musicElement.src, 'already playing');
          return;
        }

        if (musicElement) {
          musicElement.pause();
          musicElement.remove();
          musicElement = null;
        }

        if (levelIndex < 0 || levelIndex >= levelMusic.length) {
          updateMusicDebug(levelIndex, 'INVALID_LEVEL_INDEX', 'error');
          return;
        }

        const musicUrl = levelMusic[levelIndex];
        
        // If music URL is empty, don't play any music
        if (!musicUrl) {
          updateMusicDebug(levelIndex, 'No music', 'disabled');
          currentMusic = levelIndex;
          return;
        }
        
        updateMusicDebug(levelIndex, musicUrl, muted ? 'muted' : 'loading');

        if (muted) return;

        musicElement = new Audio();
        musicElement.loop = true;
        musicElement.volume = 0.3;
        musicElement.preload = 'auto';
        musicElement.crossOrigin = 'anonymous';

        musicElement.addEventListener('error', (e) => {
          updateMusicDebug(levelIndex, musicUrl, 'error: ' + e.message);
        });

        musicElement.addEventListener('canplay', () => {
          updateMusicDebug(levelIndex, musicUrl, 'ready');
        });

        musicElement.addEventListener('playing', () => {
          updateMusicDebug(levelIndex, musicUrl, 'playing');
        });

        musicElement.addEventListener('ended', () => {
          // Restart if it ends (shouldn't happen with loop=true, but just in case)
          musicElement.currentTime = 0;
          musicElement.play().catch(() => {});
        });

        musicElement.src = musicUrl;
        musicElement.load();
        
        // Try to play with a promise to handle autoplay restrictions
        const playPromise = musicElement.play();
        if (playPromise !== undefined) {
          playPromise.catch((error) => {
            updateMusicDebug(levelIndex, musicUrl, 'play failed: ' + error.message);
            // Try one more time after a short delay
            setTimeout(() => {
              if (musicElement && musicElement.paused) {
                musicElement.play().catch(() => {});
              }
            }, 100);
          });
        }
        currentMusic = levelIndex;
      }

      function clamp(v, min, max) { return Math.max(min, Math.min(max, v)); }
      function dist(a, b) { return Math.hypot(a.x - b.x, a.y - b.y); }
      function separateZombies(iterations = 2) {
        for (let step = 0; step < iterations; step++) {
          for (let i = 0; i < zombies.length; i++) {
            for (let j = i + 1; j < zombies.length; j++) {
              const a = zombies[i];
              const b = zombies[j];
              const minGap = (a.r || 17) + (b.r || 17) + 10;
              let dx = b.x - a.x;
              let dy = b.y - a.y;
              let d = Math.hypot(dx, dy);
              if (d === 0) {
                dx = 1;
                dy = 0;
                d = 1;
              }
              if (d < minGap) {
                const overlap = (minGap - d) * 0.5;
                const nx = dx / d;
                const ny = dy / d;
                a.x = clamp(a.x - nx * overlap, 20, WORLD_W - 20);
                a.y = clamp(a.y - ny * overlap, 20, WORLD_H - 20);
                b.x = clamp(b.x + nx * overlap, 20, WORLD_W - 20);
                b.y = clamp(b.y + ny * overlap, 20, WORLD_H - 20);
                walls.forEach((wall) => {
                  resolveCircleRectCollision(a, wall);
                  resolveCircleRectCollision(b, wall);
                });
              }
            }
          }
        }
      }

      function circleIntersectsRect(circle, rect) {
        const halfW = rect.w * 0.5;
        const halfH = rect.h * 0.5;
        const nearestX = clamp(circle.x, rect.x - halfW, rect.x + halfW);
        const nearestY = clamp(circle.y, rect.y - halfH, rect.y + halfH);
        const dx = circle.x - nearestX;
        const dy = circle.y - nearestY;
        return dx * dx + dy * dy < circle.r * circle.r;
      }

      function resolveCircleRectCollision(circle, rect) {
        const halfW = rect.w * 0.5;
        const halfH = rect.h * 0.5;
        const nearestX = clamp(circle.x, rect.x - halfW, rect.x + halfW);
        const nearestY = clamp(circle.y, rect.y - halfH, rect.y + halfH);
        let dx = circle.x - nearestX;
        let dy = circle.y - nearestY;
        let d2 = dx * dx + dy * dy;

        if (d2 === 0) {
          const left = Math.abs(circle.x - (rect.x - halfW));
          const right = Math.abs((rect.x + halfW) - circle.x);
          const top = Math.abs(circle.y - (rect.y - halfH));
          const bottom = Math.abs((rect.y + halfH) - circle.y);
          const minEdge = Math.min(left, right, top, bottom);
          if (minEdge === left) dx = -1;
          else if (minEdge === right) dx = 1;
          else if (minEdge === top) dy = -1;
          else dy = 1;
          d2 = 1;
        }

        const d = Math.sqrt(d2);
        const overlap = circle.r - d;
        if (overlap > 0) {
          circle.x += dx / d * overlap;
          circle.y += dy / d * overlap;
        }
      }

      function segmentIntersectsRect(x1, y1, x2, y2, rect) {
        const halfW = rect.w * 0.5;
        const halfH = rect.h * 0.5;
        const left = rect.x - halfW;
        const right = rect.x + halfW;
        const top = rect.y - halfH;
        const bottom = rect.y + halfH;
        const dx = x2 - x1;
        const dy = y2 - y1;
        let t0 = 0;
        let t1 = 1;
        const clips = [
          [-dx, x1 - left],
          [dx, right - x1],
          [-dy, y1 - top],
          [dy, bottom - y1]
        ];

        for (const [p, q] of clips) {
          if (p === 0) {
            if (q < 0) return false;
          } else {
            const r = q / p;
            if (p < 0) {
              if (r > t1) return false;
              if (r > t0) t0 = r;
            } else {
              if (r < t0) return false;
              if (r < t1) t1 = r;
            }
          }
        }
        return true;
      }

      function hasLineOfSight(from, to) {
        return !walls.some((wall) => segmentIntersectsRect(from.x, from.y, to.x, to.y, wall));
      }

      // Simple pathfinding for SNEEK ROTHSTEIN's zombies
      function findPathAroundWalls(start, target) {
        // If there's direct line of sight, just move toward target
        if (hasLineOfSight(start, target)) {
          return { x: target.x, y: target.y };
        }
        
        // Simple waypoint system: try to go around walls
        // Find the nearest wall blocking the path
        let blockingWall = null;
        let minDist = Infinity;
        
        for (const wall of walls) {
          if (segmentIntersectsRect(start.x, start.y, target.x, target.y, wall)) {
            const dist = Math.hypot(start.x - wall.x, start.y - wall.y);
            if (dist < minDist) {
              minDist = dist;
              blockingWall = wall;
            }
          }
        }
        
        if (!blockingWall) {
          return { x: target.x, y: target.y };
        }
        
        // Calculate waypoints around the wall
        const halfW = blockingWall.w * 0.5;
        const halfH = blockingWall.h * 0.5;
        
        // Determine which side of the wall is closer to start
        const leftDist = Math.abs(start.x - (blockingWall.x - halfW));
        const rightDist = Math.abs(start.x - (blockingWall.x + halfW));
        const topDist = Math.abs(start.y - (blockingWall.y - halfH));
        const bottomDist = Math.abs(start.y - (blockingWall.y + halfH));
        
        // Choose the shortest path around
        let waypointX = target.x;
        let waypointY = target.y;
        
        if (leftDist < rightDist && leftDist < topDist && leftDist < bottomDist) {
          // Go left around
          waypointX = blockingWall.x - halfW - 30;
          waypointY = start.y;
        } else if (rightDist < topDist && rightDist < bottomDist) {
          // Go right around
          waypointX = blockingWall.x + halfW + 30;
          waypointY = start.y;
        } else if (topDist < bottomDist) {
          // Go up around
          waypointX = start.x;
          waypointY = blockingWall.y - halfH - 30;
        } else {
          // Go down around
          waypointX = start.x;
          waypointY = blockingWall.y + halfH + 30;
        }
        
        return { x: waypointX, y: waypointY };
      }

                  if (player.teleportOnHit) {

                          newX < window.cageCenter.x + cageHalf + margin &&
                          newY > window.cageCenter.y - cageHalf - margin &&

                        const awayFromCageY = player.y - window.cageCenter.y;
                        const awayDist = Math.hypot(awayFromCageX, awayFromCageY) || 1;
                        newX = player.x + (awayFromCageX / awayDist) * knockback;
                        newY = player.y + (awayFromCageY / awayDist) * knockback;
                      }
                    }
                    
                    player.x = clamp(newX, 20, WORLD_W - 20);
                    player.y = clamp(newY, 20, WORLD_H - 20);

                && !swamps.some((swamp) => circleIntersectsRect(exitCircle, swamp))

      function loop(t) {
        const dt = Math.min(0.033, (t - lastTime) / 1000);
        lastTime = t;

        if (overlayTimer > 0) {

          if (overlayTimer <= 0 && queuedLevelAfterOverlay !== null) {
            currentLevel = queuedLevelAfterOverlay;
            queuedLevelAfterOverlay = null;
            startLevelNow();
          }
        }

        if (!gameState.ended) {
          update(dt);
          render();
          animationId = requestAnimationFrame(loop);
        } else if (overlayTimer > 0) {
          animationId = requestAnimationFrame(loop);
        }
      }

        return {

      function showJoystick(clientX, clientY) {
        dragState.startX = clientX;
        dragState.startY = clientY;

        dragState.knobY = clientY;
        stickBase.style.display = 'block';
        stick.style.display = 'block';

        stickBase.style.top = clientY + 'px';
        stick.style.left = clientX + 'px';

        stickBase.style.display = 'none';
        stick.style.display = 'none';

      function tryStartDrag(pointerId, clientX, clientY) {
        dragState.active = true;
        dragState.pointerId = pointerId;
        showJoystick(clientX, clientY);
        controls.dx = 0;
        controls.dy = 0;
        controls.active = false;
        return true;
      }

      function updateDrag(clientX, clientY) {
        const dx = clientX - dragState.startX;
        const dy = clientY - dragState.startY;
        const maxRadius = 52;
        const distance = Math.hypot(dx, dy) || 1;
        const clamped = Math.min(maxRadius, distance);
        const nx = dx / distance;
        const ny = dy / distance;
        const knobX = dragState.startX + nx * clamped;
        const knobY = dragState.startY + ny * clamped;

        dragState.knobX = knobX;
        dragState.knobY = knobY;
        stick.style.left = knobX + 'px';
        stick.style.top = knobY + 'px';

        const deadZone = 10;
        if (distance < deadZone) {
          controls.dx = 0;
          controls.dy = 0;
          controls.active = false;
        } else {
          controls.dx = nx;
          controls.dy = ny;
          controls.active = true;
        }
      }

      function endDrag(pointerId) {
        if (dragState.pointerId !== pointerId) return;
        dragState.active = false;
        dragState.pointerId = null;
        controls.dx = 0;
        controls.dy = 0;
        controls.active = false;
        hideJoystick();
      }
